main:

    # Διευθύνσεις πινάκων
    la $t0, A
    la $t1, B
    la $t2, C
    la $t3, MAX
    la $t4, MIN

    # Φόρτωση N
    la $t9, N
    lw $s0, 0($t9)

    li $s1, 0       # i = 0
    
    # Πρώτος έλεγχος για το Loop i
    slt $t8, $s1, $s0   

loop_i:
    # if (i >= N) break;
    beq $t8, $zero, end_i
    # [NO-NOP]: Αρχικοποίηση j=0 στο Delay Slot. Εκτελείται καθώς μπαίνουμε στο loop.
    li $s2, 0       

loop_j:
    # if (j >= N) break;
    slt $t8, $s2, $s0
    beq $t8, $zero, end_j
    # [NO-NOP]: Ξεκινάμε τον υπολογισμό i*N στο Delay Slot.
    mul $t5, $s1, $s0

    # index = (i*N + j)*4
    add $t5, $t5, $s2
    sll $t5, $t5, 2

    # load A, B, C
    addu $t6, $t0, $t5
    lw $a0, 0($t6)

    addu $t6, $t1, $t5
    lw $a1, 0($t6)

    addu $t6, $t2, $t5
    lw $a2, 0($t6)

    #### MAX ####
    addu $t7, $a0, $zero    # Max = A
    slt $t8, $t7, $a1
    beq $t8, $zero, skip1
    # [NO-NOP]: Υπολογισμός διεύθυνσης MAX ($t6) στο Delay Slot
    addu $t6, $t3, $t5      
    addu $t7, $a1, $zero    # Max = B
skip1:
    slt $t8, $t7, $a2
    beq $t8, $zero, skip2
    # [NO-NOP]: Αρχικοποίηση MIN ($t9 = A) στο Delay Slot, ενώ τελειώνουμε με το MAX
    addu $t9, $a0, $zero    
    addu $t7, $a2, $zero    # Max = C
skip2:

    #### MIN ####
    # Το Min είναι ήδη A από το προηγούμενο Delay Slot
    slt $t8, $a1, $t9
    beq $t8, $zero, skip3
    # [NO-NOP ]: Αποθηκεύουμε το MAX (sw) τώρα
    sw $t7, 0($t6)          
    addu $t9, $a1, $zero    # Min = B
skip3:
    slt $t8, $a2, $t9
    beq $t8, $zero, skip4
    # [NO-NOP]: Υπολογίζουμε τη διεύθυνση του MIN ($t6) τώρα που τελειώσαμε με το MAX
    addu $t6, $t4, $t5      
    addu $t9, $a2, $zero    # Min = C
skip4:

    # store MIN (Η διεύθυνση υπολογίστηκε στο Delay Slot του skip4)
    sw $t9, 0($t6)

    # j++ και Loop
    j loop_j
    # [NO-NOP]: Αύξηση j++ στο Delay Slot
    addi $s2, $s2, 1

end_j:
    # i++ και Loop
    addi $s1, $s1, 1
    j loop_i
    # [NO-NOP]: Έλεγχος συνθήκης (i < N) στο Delay Slot (Loop Rotation)
    slt $t8, $s1, $s0

end_i:
    break
